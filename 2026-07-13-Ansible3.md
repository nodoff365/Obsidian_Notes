## module
### Ad-hoc
```bash
#shell 명령어보단 해당 module 있으면 module 사용 지향
ansible node -m file -a "path=/test state=directory"
ansible node -m file -a "path=/test state=absent"


ansible node -m shell -a "useradd a && echo 'It1' | passwd --stdin a"
ansible node -m shell -a "tail -5 /etc/passwd"


ansible node -m user -a "name=b state=present"
ansible node -m user -a "name=b state=present password='Itt'"
ansible node -m user -a "name=a remove=true state=absent"
```

### PlayBook
```yaml
vi test.yml

---
- name: make file /test
  hosts: node
  gather_facts: false
  ignore_errors: true 
  
  tasks:
  - name: make file test
    file: 
      path: /test
      state: touch
      mode: '0777'

  - name: make directory /tdirec
    file:
      path: /tdirec
      state: directory
      mode: '0777'

ansible-playbook test.yml
ansible node -m shell -a "ls -al /"
```

```yaml
vi user.yml

---
- name: create user a & password 'It1'
  hosts: web
  gather_facts: false
  ignore_errors: true

  tasks:
  - name: create user a
    user:
      name: a
      update_password: always
#      password: 'It1' # pw 평문으로 들어감
      password: "{{ 'It1' | password_hash('sha512') }}"

  - name: delete user a & directory
    user:
      name: a
      remove: true
      state: absent

ansible-playbook user.yml

ansible web -m shell -a "tail -5 /etc/shadow"
```


```yaml
vi work.yml

---
- name: DB host user and file setup
  hosts: db
  gather_facts: false
  ignore_errors: true
  
  tasks:
  - name: create user a (uid=2000)
    user:
      name: a
      uid: 2000
      password: "{{ 'lt1' | password_hash('sha512') }}"
      state: present

  - name: create user b (uid=3000)
    user:
      name: b
      uid: 3000
      password: "{{ 'lt1' | password_hash('sha512') }}"
      state: present

  - name: create /test file
    file:
      path: /test
      state: touch
      owner: a
      group: b
      mode: 0777

  - name: create /tbabo directory
    file:
      path: /tbabo
      state: directory
      owner: a
      group: b
      mode: 0777

ansible-playbook work.yml
ansible db -m shell -a "tail -2 /etc/passwd"
ansible db -m shell -a "ls -ld /test /tbabo"
```

```yaml
vi rmwork.yml

---
- name: delete user & file
  hosts: db
  gather_facts: false
  ignore_errors: true

  tasks:
  - name: delete user
    user:
      name: "{{ item }}"
      state: absent
      remove: yes
    with_items:
      - a
      - b

  - name: delete file
    file:
      path: "{{ item }}"
      state: absent
    with_items:
      - /test
      - /tbabo
```
