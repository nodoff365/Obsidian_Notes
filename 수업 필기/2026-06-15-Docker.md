1. repository: Linux 개발회사에서 운영하는 인터넷 상의 저장소
2. 주로 package들이이곳에존재함
3. dnf or apy, zypper등의 명령어를 사용해서 다운로드함

2.package 설치 방법
2.1 source 설치(Binary설치)
2.1.1 환경설정: .configure prefix=/var
2.1.2 컴파일: make 인간이 식별하는 언어를 기계로 번역하는 작업
2.1.3 설치: make install
2.2 rpm 설치: rpm 파일 다운로드 후 설치
2.3 yum, dnf 설치: repository에서 rpm 파일 다운로드 후 설치
의존성문제까지해결
인터넷 연결, 최신버전은 아님, 커스터마이징 불가 

```bash
# rocky9-1
----------------------------------------------
# 1. Docker 설치 및 서비스 시작 (Rocky 9)
dnf -y install dnf-plugins-core                 # repo 관리 플러그인 설치
dnf config-manager --add-repo \
  https://download.docker.com/linux/centos/docker-ce.repo   # Docker 공식 repo 등록
dnf -y install docker-ce docker-ce-cli containerd.io        # (실제 설치 명령 추가 필요)
systemctl enable --now docker                   # 부팅 시 자동시작 + 즉시 시작

docker run hello-world                          # 설치 정상 동작 확인용 테스트 컨테이너


# 2. 기본 명령어
docker ps           # 실행 중인 컨테이너만 출력
docker ps -a        # 종료된 컨테이너까지 전부 출력 (ID, NAMES는 고유값)
docker rm <컨테이너> # 컨테이너 삭제 (실행 중이면 -f 필요)

docker images       # 보유한 이미지 목록
docker rmi <이미지>  # 이미지 삭제 (해당 이미지로 만든 컨테이너 없어야 함)


# 3. httpd 컨테이너 실습
docker pull httpd   # 태그 생략 시 latest 받음

# -i: 표준입력 유지, -t: 터미널 할당, -d: 백그라운드 실행
# -p 호스트포트:컨테이너포트  → 외부 60080 요청을 컨테이너 80으로 전달
docker run -itd -p 60080:80 --name h1 httpd     # 접속: 10.0.0.11:60080
docker run -itd -p 60180:80 --name h2 httpd
docker run -itd -p 60280:80 --name h3 httpd
docker run -itd -p 60380:80 --name h4 httpd

docker ps -aq                       # 컨테이너 ID만 출력 (-q = quiet)
docker rm -f $(docker ps -aq)       # 모든 컨테이너 강제 삭제 (-f로 실행 중인 것도 종료 후 삭제)


# 4. 다양한 이미지 실습
docker pull alpine                  # 테스트용, 경량 리눅스
docker pull nginx
docker pull busybox                 # 테스트용, 초경량 유틸 모음
docker pull rockylinux/rockylinux

docker run -itd -p 65080:80 --name n1 nginx
docker run -itd -p 65180:80 --name n2 nginx
docker run -itd -p 65280:80 --name n3 nginx
docker run -itd -p 65380:80 --name n4 nginx
docker run -itd -p 65480:80 --name n5 nginx
docker rm -f $(docker ps -aq)

docker run hello-world
docker run --name he1 hello-world
docker ps -a

docker search nginx # 기본 25줄 출력
docker search nginx --limit 10
docker search -f stars=50 nginx
docker search -f is-official=true nginx
docker search nginx --no-trunc

------------------------------------------
# 컨테이너 생명주기
# 1. create: 생성만 (실행 X), STATUS = Created
docker create -it --name a1 alpine
docker start a1 # 시작 → -it라서 sh 유지, STATUS = Up
docker ps -a

docker create -it --name a2 alpine
docker start a2
docker ps -a # a2도 Up


# 2. stop / start: 중지했다 다시 시작
docker stop a2  # STATUS Exited
docker start a2 # 다시 Up


# 3. pause / unpause: 일시정지 (프로세스는 메모리에 유지, 멈춤만)
docker pause a2 # STATUS Paused
docker unpause a2 # 다시 Up


# 4. 정리
docker stop a1 a2
docker rm a1 a2 # 삭제 (실행 중이면 stop 먼저, 또는 rm -f)


# 5. run + 셸 진입, 빠져나오는 두 방식
# (A) exit 방식 → 컨테이너도 꺼짐
docker run -it --name a1 alpine   # 생성+실행+셸 접속 (default CMD = /bin/sh)
  # ↓ 컨테이너 셸 안에서
  hostname        # 호스트명(=컨테이너 ID 앞자리) 출력
  exit            # 셸(메인 프로세스) 종료 → 컨테이너도 Exited
docker ps -a      # a1: Exited 확인


# (B) detach 방식 → 컨테이너는 켜둔 채 빠져나옴
docker run -it --name a2 alpine
  # ↓ 셸 안에서
  hostname
  # 여기서 Ctrl + P 누른 뒤, 이어서 Ctrl + Q
  #  → 셸은 안 끄고 터미널만 빠져나옴 (detach)
docker ps -a      # a2: Up 유지 확인

------------------------------------------------
docker run -itd --name n1 nginx
docker exec -it n1 /bin/bash

docker inspect n1

docker run -itd --name a1 alpine
docker attach a1
docker exec -it a1 /bin/sh #exec 실행은 exit로 빠져나와도됨
docker exec a1 ip a
```
