# MySQL 학생 성적과 주소록 관리 데이터베이스
---
데이터베이스 : student
테이블 : sungjuk / 컬럼 : 번호, 이름, 국어, 영어, 수학
테이블 : jusorok / 컬럼 : 번호, 이름, 주소
- 데이터 10개 입력
- sungjuk 테이블 데이터의 총점과 평균, 순위를 계산하고, 순위가 있는 테이블 sungjuk_ranking을 만들고,
- 컬럼명은 번호, 이름, 국어, 영어, 수학, 총점, 평균, 순위
```sql
/* 0) DB 생성/선택 */

CREATE DATABASE IF NOT EXISTS student
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_0900_ai_ci;
USE student;
  
/* 1) 테이블 생성 */
DROP TABLE IF EXISTS sungjuk_ranking;
DROP TABLE IF EXISTS sungjuk;
DROP TABLE IF EXISTS jusorok;
  
CREATE TABLE sungjuk (
  `번호` INT NOT NULL,
  `이름` VARCHAR(20) NOT NULL,
  `국어` TINYINT UNSIGNED NOT NULL,
  `영어` TINYINT UNSIGNED NOT NULL,
  `수학` TINYINT UNSIGNED NOT NULL,
  PRIMARY KEY (`번호`)
) ENGINE=InnoDB;
  
CREATE TABLE jusorok (
  `번호` INT NOT NULL,
  `이름` VARCHAR(20) NOT NULL,
  `주소` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`번호`)
) ENGINE=InnoDB;
  
/* 2) 데이터 10개 입력 */
INSERT INTO sungjuk (`번호`,`이름`,`국어`,`영어`,`수학`) VALUES
(1,'김민준',92,88,95),
(2,'이서연',85,90,87),
(3,'박지훈',78,82,80),
(4,'최유진',96,94,91),
(5,'정현우',88,76,84),
(6,'한지민',91,93,89),
(7,'오세훈',69,74,72),
(8,'장하늘',83,81,86),
(9,'윤서아',95,97,96),
(10,'신도윤',80,85,79);
 
INSERT INTO jusorok (`번호`,`이름`,`주소`) VALUES
(1,'김민준','서울특별시 강남구 테헤란로 123'),
(2,'이서연','서울특별시 마포구 월드컵북로 45'),
(3,'박지훈','인천광역시 연수구 송도과학로 12'),
(4,'최유진','대전광역시 유성구 대학로 99'),
(5,'정현우','대구광역시 수성구 동대구로 210'),
(6,'한지민','부산광역시 해운대구 센텀중앙로 8'),
(7,'오세훈','광주광역시 북구 첨단과기로 77'),
(8,'장하늘','울산광역시 남구 삼산로 55'),
(9,'윤서아','세종특별자치시 도움3로 20'),
(10,'신도윤','경기도 성남시 분당구 판교로 256');
  
/* 3) 총점/평균/순위 계산해서 sungjuk_ranking 테이블 생성
      - 총점: 국어+영어+수학
      - 평균: (총점/3) 소수 둘째자리
      - 순위: 총점 내림차순 기준 (동점은 같은 순위, 다음 순위는 건너뜀: RANK)
*/
CREATE TABLE sungjuk_ranking AS
SELECT
  s.`번호`,
  s.`이름`,
  s.`국어`,
  s.`영어`,
  s.`수학`,
  (s.`국어` + s.`영어` + s.`수학`) AS `총점`,
  ROUND((s.`국어` + s.`영어` + s.`수학`) / 3, 2) AS `평균`,
  RANK() OVER (
    ORDER BY (s.`국어` + s.`영어` + s.`수학`) DESC, s.`이름` ASC
  ) AS `순위`
FROM sungjuk s;

/* (선택) sungjuk_ranking에 PK 추가하고 싶다면 */
ALTER TABLE sungjuk_ranking
  ADD PRIMARY KEY (`번호`);
  
/* 확인 */
SELECT * FROM sungjuk_ranking ORDER BY `순위`, `번호`;
```
