```bash
# 네임스페이스
kubectl get pods

kubectl get namespace # 네임스페이스 확인
kubectl get ns # kubectl api-resources 에서 별명 확인

delete namespace default # 기본 생성 네임스페이스 삭제 불가
kubectl get pods --namespace kube-system # kube-system 네임스페이스에 있는 파드들만 보여줌

# 팀별로 네임스페이스 사용 가능(리소스 격리 공간)
# 지정안하면 기본적으로 default에 생성
kubectl create namespace 1team
kubectl run app-nginx --image nginx --port 80 --namespace 1team #1team 네임스페이스에 nginx 생성
kubectl get pods --namespace 1team
kubectl delete pods app-nginx --namespace 1team
kubectl delete namespace 1team
```

# kubectl api-resources / apiVersion 확인

vi ns1.yml

apiVersion: v1
kind: Namespace
metadata:
  name: 2team
  labels:
    env: devel

kubectl apply -f ns1.yml # ns 1team, 2team 생성
