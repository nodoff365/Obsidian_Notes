### HTTPS 지원을 위한 OpenSSL 기반 SSL/TLS 설정(Self-Signed)
---
##### OpenSSL 버전 및 FIPS 상태 확인
```bash
cat /proc/sys/crypto/fips_enabled
# 1 → FIPS 활성
# 0 → 비활성
```

```bash
cat > san.cnf <<'EOF'

[req]
default_bits       = 4096
prompt             = no
default_md         = sha256
distinguished_name = dn
x509_extensions    = v3_req
 
[dn]
C=KR
ST=Gyeonggi
L=Paju
O=KCCI
OU=Security
CN=web.kcci.edu
  
[v3_req]
subjectAltName = @alt_names
  
[alt_names]
DNS.1 = web.kcci.edu
IP.1  = 192.168.200.21

EOF
#ip랑 dns 일치하는지 확인
#ip a
#hostname
```


```bash
sudo mkdir -p /etc/pki/nginx/private
sudo chmod 700 /etc/pki/nginx/private

sudo openssl req -x509 -nodes -days 365 \                 #인증서 생성
  -newkey rsa:4096 \
  -keyout /etc/pki/nginx/private/server.key \
  -out /etc/pki/nginx/server.crt \
  -config san.cnf
  
sudo chmod 600 /etc/pki/nginx/private/server.key
sudo chown -R root:root /etc/pki/nginx
  
sudo restorecon -v \
  /etc/pki/nginx/private/server.key \
  /etc/pki/nginx/server.crt
  
# SAN 포함 여부 반드시 확인
openssl x509 -in /etc/pki/nginx/server.crt -noout -text | grep -A2 "Subject Alternative Name"
  
sudo nginx -t
sudo systemctl restart nginx
```


### Nginx HTTPS + OpenSSL 설정 예제 (권장)
```bash
sudo tee /etc/nginx/conf.d/ssl.conf >/dev/null <<'EOF'

server {
    listen 443 ssl;
    listen [::]:443 ssl;
    http2 on;
  
    server_name web.kcci.edu;
    root /usr/share/nginx/html;
  
    ssl_certificate     /etc/pki/nginx/server.crt;
    ssl_certificate_key /etc/pki/nginx/private/server.key;
  
    # RHEL 10 crypto-policy 연동(핵심)
    ssl_ciphers PROFILE=SYSTEM;
    ssl_prefer_server_ciphers on;
  
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
  
    add_header Strict-Transport-Security "max-age=31536000" always;
  
    include /etc/nginx/default.d/*.conf;
}
EOF


sudo systemctl restart nginx
```
