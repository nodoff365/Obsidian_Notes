## Docker Network
```bash
docker run -itd --name a1 alpine
docker ps
docker exec a1 ip a
ip a
```

```bash
docker network ls
NETWORK ID     NAME      DRIVER    SCOPE
5dffb5ceedea   bridge    bridge    local
5adce4c150ba   host      host      local
ac1b2020df36   none      null      local
# bridge 는 방화벽 자동으로 열림
# host 는 방화벽 자동으로 안열림

docker exec a1 ping 172.17.0.3
PING 172.17.0.3 (172.17.0.3): 56 data bytes
64 bytes from 172.17.0.3: seq=0 ttl=64 time=0.187 ms
64 bytes from 172.17.0.3: seq=1 ttl=64 time=0.049 ms
64 bytes from 172.17.0.3: seq=2 ttl=64 time=0.045 ms
#같은 네트워크라 ping 가능

docker rm -f $(docker ps -aq)
```

```bash
docker inspect bridge
```
