##### 1.6 Sudo 명령 그룹 생성

```bash
# 이름이 allow_all, allow_admins인 Sudo Rule 생성
ipa sudorule-add allow_admins --desc="Admin full sudo access"

# admin 그룹에 속한 사용자에게 적용
ipa sudorule-add-user allow_admins --groups={admins,nova,professor}

ipa sudocmd-add ALL --desc="All commands"

# all_hosts 호스트그룹에 속한 모든 서버에 적용 (전체 서버에 적용)
ipa sudorule-add-host allow_admins --hostgroups=all_hosts

# 모든 명령 실행 허용
ipa sudorule-add-allow-command allow_admins --sudocmds=ALL

# 패스워드 없이 실행 (NOPASSWD 효과)
# IPA SSSD가 /etc/sudoers.d/sssd 로 변환 시 내부적으로:
# %admin ALL=(ALL) NOPASSWD: ALL
ipa sudorule-add-option allow_admins --sudooption='!authenticate'
```

```bash
# 룰 정책 목록 확인
ipa sudorule-find
ipa sudorule-show web_ops
ipa sudorule-show web_ops --all
ipa sudorule-show allow_admins

# 특정 sudo rule 삭제
ipa sudorule-del allow-admins
  
# 특정 명령 삭제
ipa sudocmd-del /usr/bin/mysql
  
# 명령 그룹 삭제
ipa sudocmdgroup-del web_commands
  
# 특정 그룹만 rule 제거 ex) allow_admins에서 professor만 제거
ipa sudorule-remove-user allow_admins --groups=professor
# 특정 그룹 추가
ipa sudorule-add-user allow_admins --groups=professor 
  
# 특정 호스트만 제거
ipa sudorule-remove-host allow_admins --hostgroups=web_servers
# 특정 호스트 hg_web,hg_was 추가
ipa sudorule-add-host allow_admins --hostgroups={hg_web,hg_was}
 
# 특정 명령만 제거
ipa sudorule-remove-allow-command allow_admins --sudocmds=/usr/bin/mysql
```
