```bash
# ACL 조회
getfacl <파일/디렉토리>

# 예시
getfacl file.txt
getfacl /home/user

# ACL 설정 (사용자에게 읽기/쓰기/실행 권한)
setfacl -m u:<사용자명>:rwx <파일/디렉토리>

# 예시
setfacl -m u:john:rwx file.txt
setfacl -m u:sgm:rx /web

# ACL 설정 (그룹에게 권한)
setfacl -m g:<그룹명>:rwx <파일/디렉토리>

# 예시
setfacl -m g:developers:rwx file.txt

# ACL 제거
setfacl -x u:<사용자명> <파일/디렉토리>

# 예시
setfacl -x u:john file.txt

# 디렉토리와 하위 항목에 재귀 적용
setfacl -R -m u:<사용자명>:rwx <디렉토리>

# 예시
setfacl -R -m u:john:rwx /web

# ACL 모두 제거
setfacl -b <파일/디렉토리>
```

**권한 기호**

- `r` = 읽기 (read)
- `w` = 쓰기 (write)
- `x` = 실행 (execute)

**주요 옵션**

- `-m` = 수정 (modify)
- `-x` = 제거 (remove)
- `-b` = 초기화 (remove all)
- `-R` = 재귀 (recursive)

```bash
# 1. 최상위 디렉토리 생성
mkdir /gogle
mkdir /gogle/admin
mkdir /gogle/sales

# 2. 사용자 4개 생성
useradd a
useradd b
useradd c
useradd d

# 3. ACL 설정

# admin 팀원 (a): admin 읽기만
setfacl -m u:a:rx /gogle/admin

# admin 팀장 (b): admin 읽기/쓰기, sales 읽기
setfacl -m u:b:rwx /gogle/admin
setfacl -m u:b:rx /gogle/sales

# sales 팀원 (c): sales 읽기만
setfacl -m u:c:rx /gogle/sales

# sales 팀장 (d): sales 읽기/쓰기, admin 읽기
setfacl -m u:d:rwx /gogle/sales
setfacl -m u:d:rx /gogle/admin

# 4. 기본 권한 제거 (owner만 접근)
chmod 700 /gogle/admin
chmod 700 /gogle/sales

# 5. ACL 확인
getfacl /gogle/admin
getfacl /gogle/sales

# 6. 테스트 파일 생성
echo "admin file" > /gogle/admin/admin.txt
echo "sales file" > /gogle/sales/sales.txt

# 7. 읽기 권한 테스트

# a: admin만 접근 가능 (팀원)
sudo -u a ls /gogle/admin         # admin.txt 보임
sudo -u a ls /gogle/sales         # Permission denied

# b: admin, sales 둘 다 접근 가능 (팀장)
sudo -u b ls /gogle/admin         # 파일 보임
sudo -u b ls /gogle/sales         # 파일 보임

# c: sales만 접근 가능 (팀원)
sudo -u c ls /gogle/admin         # Permission denied
sudo -u c ls /gogle/sales         # sales.txt 보임

# d: admin, sales 둘 다 접근 가능 (팀장)
sudo -u d ls /gogle/admin         # 파일 보임
sudo -u d ls /gogle/sales         # 파일 보임

# 8. 쓰기 권한 테스트

# b: admin에만 쓰기 가능 (admin 팀장)
sudo -u b touch /gogle/admin/b_created.txt    # 성공
sudo -u b touch /gogle/sales/b_fail.txt       # Permission denied

# d: sales에만 쓰기 가능 (sales 팀장)
sudo -u d touch /gogle/sales/d_created.txt    # 성공
sudo -u d touch /gogle/admin/d_fail.txt       # Permission denied

# a: 읽기만 가능, 쓰기 불가 (admin 팀원)
sudo -u a touch /gogle/admin/a_fail.txt       # Permission denied

# c: 읽기만 가능, 쓰기 불가 (sales 팀원)
sudo -u c touch /gogle/sales/c_fail.txt       # Permission denied

# 9. 최종 ACL 확인
getfacl -R /gogle
```
