```bash
# rocky9-1
ansible node -m ping

ansible node -m shell -a "mkdir /test"
ansible node -m shell -a "ls -al /"
ansible node -m shell -a "rm -rf /test"
ansible node -m shell -a "ls -al /"

mkdir /shell
cd /shell/

vi ls.yml
---
- name: ls 디렉토리 확인
  hosts: node
  gather_facts: false
  ignore_errors: true

  tasks:
  - name: view / directory
    shell: ls -al /

ansible-playbook -i /etc/ansible/hosts ls.yml
ansible-playbook -i /etc/ansible/hosts ls.yml -vv # 상세 확인

vi md.yml
---
- name: make directory
  hosts: node
  gather_facts: false
  ignore_errors: true

  tasks:
  - name: make directory /test
    shell: mkdir /test

  - name: view directory
    shell: ls -al / | grep test
   
  - name: remove directory /test
    shell: rm -rf /test
  
ansible-playbook md.yml
```
