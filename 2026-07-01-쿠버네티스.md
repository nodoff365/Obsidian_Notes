## Pod 생성 방법 실습 - 명령형 vs 선언형
```bash
# 방법 1: kubectl run으로 바로 생성 (명령형)
kubectl run nginx --image nginx --port=80 --image-pull-policy=Never
kubectl get po
# NAME    READY   STATUS    RESTARTS   AGE
# nginx   1/1     Running   0          13s

# 대화형으로 컨테이너 접속하며 생성
kubectl run alpine -it --image=alpine -- sh

# --dry-run으로 yml 파일만 생성 (실제 배포 X, 템플릿 뽑기용)
kubectl run httpd --image=httpd --dry-run=client -o yaml > httpd.yml
```

```bash
# 방법 2: yml 파일 작성 후 생성 (선언형)
vi al.yml

apiVersion: v1
kind: Pod
metadata:
  name: alpine
  labels:
    env: prod
spec:
  containers:
  - name: alpine
    image: alpine
    imagePullPolicy: Never
    command: ["tail", "-F", "/test"]

# dry-run=server로 서버 측 검증 (실제 생성 전 유효성 확인)
kubectl apply -f al.yml --dry-run=server

# 실제 생성
kubectl apply -f al.yml

# 삭제 방법 2가지
kubectl delete -f al.yml            # yml 파일 기준 삭제
kubectl delete po alpine1 alpine2   # Pod 이름 직접 지정 삭제
```

```bash
# [멀티 컨테이너 Pod 실습]
# 하나의 Pod 안에 컨테이너 2개(alpine, nginx)를 배치하고
# 각 컨테이너에 개별 접속하는 방법을 확인하는 실습
vi alng.yml

apiVersion: v1
kind: Pod
metadata:
  name: alng
  labels:
    env: prod
    app: nginx
spec:
  containers:
  - name: a1
    image: alpine
    imagePullPolicy: IfNotPresent
    command: ["tail", "-F", "/test"]
  - name: n1
    image: nginx
    imagePullPolicy: IfNotPresent
    ports:
    - containerPort: 80

kubectl apply -f alng.yml

# 상태 확인 - 한 Pod에 두 컨테이너가 모두 실행되는지 검증
kubectl get po
# NAME    READY   STATUS    RESTARTS   AGE
# alng    2/2     Running   0          5s
# READY 2/2 = 컨테이너 2개 중 2개 실행 중
--------------------------------------------
# [특정 컨테이너 접속 실습]
# 멀티 컨테이너 Pod에서 원하는 컨테이너를 골라 접속하는 방법 확인

# 기본 접속 - 컨테이너 미지정 시 첫 번째 컨테이너(a1)에 자동 접속
kubectl exec -it alng -- sh
# 출력: Defaulted container "a1" out of: a1, n1

# 특정 컨테이너 접속 - -c 옵션으로 컨테이너(n1) 지정
kubectl exec -it alng -c n1 -- bash

# nginx 컨테이너의 웹 파일 경로 확인 (컨테이너별로 내용이 다름을 확인)
ls /usr/share/nginx/html/
# 50x.html  index.html
```

## WordPress/Mysql
```bash
kubectl run mysql --image=mysql:8.0 --port=3306 --image-pull-policy=Never \
--env="MYSQL_ROOT_PASSWORD=It12345!" \
--env="MYSQL_DATABASE=word" \
--env="MYSQL_USER=sgm" \
--env="MYSQL_PASSWORD=It12345!"

kubectl get po -o wide
# NAME    READY   STATUS    RESTARTS   AGE   IP              NODE
# mysql   1/1     Running   0          14m   192.168.104.5   node2

kubectl exec -it mysql -- bash
mysql -u root -p
# 비밀번호: It12345!

# word 데이터베이스가 생성됐는지 확인

--env="WORDPRESS_DB_HOST=192.168.104.5" \
--env="WORDPRESS_DB_USER=root" \
--env="WORDPRESS_DB_PASSWORD=It12345!" \
--env="WORDPRESS_DB_NAME=word"

# NAME    READY   STATUS    RESTARTS   AGE   IP              NODE
# mysql   1/1     Running   0          18m   192.168.104.5   node2
# word    1/1     Running   0          48s   192.168.135.8   node3

dnf install -y lynx
lynx 192.168.135.8
# word Pod의 IP로 워드프레스 웹 페이지 접속 확인

kubectl expose --name=wordsvc pod word --type=NodePort
```
