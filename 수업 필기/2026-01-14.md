```bash
mkdir /samba

# 설정 확인

groupadd users
useradd -d /home/users/kcci -s /bin/bash -g users -u 1111 kcci
passwd kcci

smbpasswd -a kcci
# Samba 사용자 계정을 추가하고 비밀번호를 설정

tee -a /etc/samba/smb.conf <<EOF
[samba]
    path = /samba
    browseable = yes
    writable = yes
    guest ok = no
    valid users = kcci
EOF

# /samba 디렉터리를 ‘kcci 사용자만’ 읽고, 쓸 수 있는 인증 기반 Samba 공유로 만듦
```

# NFS(Network File System) 서버 설정 (공유 폴더: /public)
---
### NFS 서버 설정
##### ns.kcci.edu (192.168.200.23) 서버를 NFS 서버로 가정하고 진행

1. 서버 측 설정 (ns.kcci.edu)
```bash
mkdir /public
chmod 777 /public

setsebool -P nfs_export_all_rw on
# SELinux NFS 허용 불리언 설정
getsebool -a | grep nfs_export_all_rw
# SELinux NFS 설정 확인

echo "/public 192.168.200.0/24(rw,sync,no_root_squash)" | sudo tee -a /etc/exports
# NFS 내보내기 설정

cat /etc/exports
# NFS 설정 확인
# /public  192.168.200.0/24(rw,sync,no_root_squash)

exportfs -ra

exportfs -v # 현재 정상적으로 공유 중인 목록 확인
결과: /public         192.168.200.0/24(sync,wdelay,hide,no_subtree_check,sec=sys,rw,secure,no_root_squash,no_all_squash)


sudo systemctl enable --now nfs-server
sudo firewall-cmd --permanent --add-service=nfs
sudo firewall-cmd --reload
firewall-cmd --list-all
```

2. 클라이언트 설정

```bash
sudo mount -t nfs 192.168.200.23:/public /nfs_public
```
