##### VNet + Subnet 생성
- 생성
```bash
az network vnet create \
  --resource-group st613-rg-lab-seoul \
  --name vnet-lab \
  --location koreacentral \
  --address-prefix 10.10.0.0/16 \
  --subnet-name snet-web \
  --subnet-prefix 10.10.1.0/24
```
- 확인
```bash
# vnet(vnet-lab)의 subnet(snet-web) 확인

az network vnet subnet list \
  -g st613-rg-lab-seoul \
  --vnet-name vnet-lab \
  -o table
```

##### 공인 IP 생성 (VM 외부 접속용)
```bash
az network public-ip create \
  --resource-group st613-rg-lab-seoul \
  --name pip-vm01 \
  --sku Standard \
  --allocation-method Static
```

##### NIC (Network Interface) 생성
```bash
#물리적인 네트워크(NIC) 생성(nic-vm01)
az network nic create \
  --resource-group st613-rg-lab-seoul \
  --name nic-vm01 \
  --vnet-name vnet-lab \
  --subnet snet-web \
  --public-ip-address pip-vm01
```

##### NSG 생성 및 규칙 추가 (SSH 허용)
```bash
#NSG(Network Security Group) 생성(nsg-vm)
az network nsg create \
  --resource-group st613-rg-lab-seoul \
  --name nsg-vm01

#NSG rule 생성
az network nsg rule create \
  --resource-group st613-rg-lab-seoul \
  --nsg-name nsg-vm01 \
  --name allow-ssh \
  --priority 100 \
  --direction Inbound \
  --access Allow \
  --protocol Tcp \
  --destination-port-ranges 22

az network nsg rule create \
  --resource-group st613-rg-lab-seoul \
  --nsg-name nsg-vm01 \
  --name allow-http \
  --priority 200 \
  --direction Inbound \
  --access Allow \
  --protocol Tcp \
  --destination-port-ranges 80

az network nsg rule create \
  --resource-group st613-rg-lab-seoul \
  --nsg-name nsg-vm01 \
  --name allow-https \
  --priority 300 \
  --direction Inbound \
  --access Allow \
  --protocol Tcp \
  --destination-port-ranges 443
```

##### NIC에 NSG 연결
```bash
# 물리적인 NIC(nic-vm01)에 NSG(nsg-vm) 연결(추가)
az network nic update \
  --resource-group st613-rg-lab-seoul \
  --name nic-vm01 \
  --network-security-group nsg-vm01
```

##### VM 생성 (Linux 예제 – RHEL 기준)

##### 자동화 All-in-One 스크립트(/home/nova/Work/Azure_CLI/all-in-one.sh)
```bash
#!/bin/bash
RG=st613-rg-allinone-seoul
LOC=koreacentral
VNET=vnet-lab
SUBNET=snet-web
PIP=pip-vm01
NIC=nic-vm01
NSG=nsg-vm01
VM=rhel9-vm01
IMAGE=RedHat:RHEL:9-lvm-gen2:latest
ADMINUSER=azureuser

# 리소스 그룹 생성(st601-rg-lab-seoul)
az group create -n $RG -l $LOC
  
# VNet + Subnet 생성
az network vnet create \
  -g $RG -n $VNET \
  --address-prefix 10.10.0.0/16 \
  --subnet-name $SUBNET \
  --subnet-prefix 10.10.1.0/24

# 공인 IP 생성 (VM 외부 접속용)
az network public-ip create \
  --resource-group $RG \
  --name $PIP \
  --sku Standard \
  --allocation-method Static

# 물리적인 네트워크(NIC) 생성(nic-vm01)
az network nic create \
  --resource-group $RG \
  --name $NIC \
  --vnet-name $VNET \
  --subnet $SUBNET \
  --public-ip-address $PIP

# NSG(Network Security Group) 생성(nsg-vm)
az network nsg create \
  --resource-group $RG \
  --name $NSG
  
# NSG rule 생성
az network nsg rule create \
  --resource-group $RG \
  --nsg-name $NSG \
  --name allow-ssh \
  --priority 100 \
  --direction Inbound \
  --access Allow \
  --protocol Tcp \
  --destination-port-ranges 22
  

# 물리적인 NIC(nic-vm01)에 NSG(nsg-vm) 연결(추가)
az network nic update \
  --resource-group $RG \
  --name $NIC \
  --network-security-group $NSG

az vm create \
  --resource-group $RG \
  --name $VM \
  --location $LOC \
  --nics $NIC \
  --image $IMAGE \
  --size Standard_B2s \
  --admin-username $ADMINUSER \
  --authentication-type ssh \
  --generate-ssh-keys
```
