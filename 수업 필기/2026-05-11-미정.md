## 1. Wordpress download
```bash
# rocky9-2~3 wordpress 설치 및 설정 
dnf install -y wget tar httpd php php-mysqlnd php-gd php-cli

# https://ko.wordpress.org/download/ 접속
# 릴리즈 - tar.gz(우클릭) - 링크주소복사
wget https://ko.wordpress.org/wordpress-6.9.4-ko_KR.tar.gz
tar xfvz wordpress-6.9.4-ko_KR.tar.gz

ls wordpress
cp -ar wordpress/* /var/www/html/ #메타데이터, 디렉토리 포함 복사
sed -i "s/DirectoryIndex index.html/DirectoryIndex index.php/g" /etc/httpd/conf/httpd.conf  # 시작 페이지를 index.php로 변경

cp /var/www/html/{wp-config-sample.php,wp-config.php}
vi wp-config.php # 내용 변경 해줘야함 # 변경 후 데이터베이스 커넥션 오류 나와야함
sed -i "s/database_name_here/wordpress/g" /var/www/html/wp-config.php
sed -i "s/username_here/sgm/g" /var/www/html/wp-config.php
sed -i "s/password_here/It12345@/g" /var/www/html/wp-config.php
sed -i "s/localhost/10.0.0.14/g" /var/www/html/wp-config.php

mv /etc/httpd/conf.d/{welcome.conf,welcome.conf.bak}
systemctl enable --now httpd
firewall-cmd --permanent --add-port=80/tcp
firewall-cmd --reload

# 로드밸런싱 추후 확인 위해 작성  
echo $HOSTNAME > /var/www/html/health.html
#10.0.0.12/health.html
#10.0.0.13/health.html
#제대로 작동하는지 확인 
```

## 2. 스크립트
`스크립트 작성시 sed, awk`

```bash
# wordpress

#!/bin/bash
dnf install -y wget tar httpd php php-cli php-gd php-mysqlnd
wget https://ko.wordpress.org/wordpress-6.9.4-ko_KR.tar.gz
tar xfvz wordpress-6.9.4-ko_KR.tar.gz
cp -ar wordpress/* /var/www/html/
sed -i "s/DirectoryIndex index.html/DirectoryIndex index.php/g" /etc/httpd/conf/httpd.conf
cp /var/www/html/{wp-config-sample.php,wp-config.php}
sed -i "s/database_name_here/wordpress/g" /var/www/html/wp-config.php
sed -i "s/username_here/sgm/g" /var/www/html/wp-config.php
sed -i "s/password_here/It12345@/g" /var/www/html/wp-config.php
sed -i "s/localhost/10.0.0.14/g" /var/www/html/wp-config.php
mv /etc/httpd/conf.d/{welcome.conf,welcome.conf.bak}
echo $HOSTNAME > /var/www/html/health.html
systemctl enable --now httpd
firewall-cmd --permanent --add-port=80/tcp
firewall-cmd --reload
```

```bash
# mysql

#!/bin/bash
dnf install -y mysql-server
systemctl enable --now mysqld
firewall-cmd --permanent --add-port=3306/tcp
firewall-cmd --reload
mysql -uroot -e "create user 'sgm'@'%' identified by 'It12345@'; grant all privileges on *.* to 'sgm'@'%'; flush privileges;"
```

```bash
# haproxy

#! /bin/bash
dnf install -y haproxy
sed -i "s/*:5000/*:80/g" /etc/haproxy/haproxy.cfg
sed -i "s/use_backend static/use_backend app/g" /etc/haproxy/haproxy.cfg
sed -i "s/server  app3/#server  app3/g" /etc/haproxy/haproxy.cfg
sed -i "s/server  app4/#server  app4/g" /etc/haproxy/haproxy.cfg
sed -i "s/127.0.0.1:5001/10.0.0.12:80/g" /etc/haproxy/haproxy.cfg
sed -i "s/127.0.0.1:5002/10.0.0.13:80/g" /etc/haproxy/haproxy.cfg
systemctl enable --now haproxy
firewall-cmd --permanent --add-port=80/tcp
firewall-cmd --reload
```
